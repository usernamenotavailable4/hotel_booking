-- -----------------------------
-- LOCATION & ADDRESS
-- -----------------------------
CREATE TABLE country (
  id INT AUTO_INCREMENT PRIMARY KEY,
  iso_code VARCHAR(8) NOT NULL,
  name VARCHAR(200) NOT NULL,
  UNIQUE KEY ux_country_iso (iso_code)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE region (
  id INT AUTO_INCREMENT PRIMARY KEY,
  country_id INT NOT NULL,
  name VARCHAR(200) NOT NULL,
  FOREIGN KEY (country_id) REFERENCES country(id) ON DELETE CASCADE,
  INDEX (country_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE city (
  id INT AUTO_INCREMENT PRIMARY KEY,
  region_id INT NOT NULL,
  name VARCHAR(150) NOT NULL,
  lat DECIMAL(9,6) DEFAULT NULL,
  lng DECIMAL(9,6) DEFAULT NULL,
  timezone VARCHAR(64) DEFAULT NULL,
  FOREIGN KEY (region_id) REFERENCES region(id) ON DELETE cascade,
  INDEX (region_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE address (
  id INT AUTO_INCREMENT PRIMARY KEY,
  street VARCHAR(255),
  city_id INT,
  postal_code VARCHAR(20),
  lat DECIMAL(9,6) DEFAULT NULL,
  lng DECIMAL(9,6) DEFAULT NULL,
  FOREIGN KEY (city_id) REFERENCES city(id) ON DELETE SET NULL,
  INDEX (city_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- -----------------------------
-- USERS & PROFILES
-- -----------------------------
CREATE TABLE `user` (
  user_id INT AUTO_INCREMENT PRIMARY KEY,
  email VARCHAR(255) NOT NULL UNIQUE,
  password_hash VARCHAR(255) NOT NULL,
  role VARCHAR(64) DEFAULT 'customer',
  is_active TINYINT(1) DEFAULT 1,
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE user_profile (
  user_id INT PRIMARY KEY,
  first_name VARCHAR(150),
  last_name VARCHAR(150),
  phone VARCHAR(50) UNIQUE,
  dob DATE DEFAULT NULL,
  preferred_language VARCHAR(10),
  timezone VARCHAR(64),
  FOREIGN KEY (user_id) REFERENCES `user`(user_id) ON DELETE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- -----------------------------
-- HOTEL & RELATED
-- -----------------------------
CREATE TABLE hotel (
  id INT AUTO_INCREMENT PRIMARY KEY,
  name VARCHAR(255) NOT NULL,
  slug VARCHAR(255) DEFAULT NULL,
  star_rating TINYINT DEFAULT NULL,
  address_id INT DEFAULT NULL,
  main_phone VARCHAR(50),
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  status VARCHAR(32) DEFAULT 'draft',
  description TEXT,
  checkin_time TIME DEFAULT NULL,
  checkout_time TIME DEFAULT NULL,
  language_support VARCHAR(255) DEFAULT NULL,
  FOREIGN KEY (address_id) REFERENCES address(id) ON DELETE SET NULL,
  INDEX (address_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- -----------------------------
-- ROOMS & AMENITIES
-- -----------------------------
CREATE TABLE room_type (
  id INT AUTO_INCREMENT PRIMARY KEY,
  hotel_id INT NOT NULL,
  code VARCHAR(64),
  name VARCHAR(255),
  max_adults INT DEFAULT 2,
  max_children INT DEFAULT 0,
  description TEXT,
  price_per_night int(10),
  FOREIGN KEY (hotel_id) REFERENCES hotel(id) ON DELETE CASCADE,
  INDEX (hotel_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE room (
  id INT AUTO_INCREMENT PRIMARY KEY,
  room_type_id INT NOT NULL,
  room_number VARCHAR(64),
  floor VARCHAR(20),
  status VARCHAR(32) DEFAULT 'available',
  FOREIGN KEY (room_type_id) REFERENCES room_type(id) ON DELETE CASCADE,
  INDEX (room_type_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE amenity (
  id INT AUTO_INCREMENT PRIMARY KEY,
  name VARCHAR(255) NOT NULL,
  description TEXT,
  amenity_type VARCHAR(32) DEFAULT 'hotel'  -- 'hotel' or 'room'
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- polymorphic amenity_map: entity_type = 'hotel'|'room_type'
CREATE TABLE amenity_map (
  id INT AUTO_INCREMENT PRIMARY KEY,
  amenity_id INT NOT NULL,
  entity_type VARCHAR(32) NOT NULL,
  entity_id INT NOT NULL,
  FOREIGN KEY (amenity_id) REFERENCES amenity(id) ON DELETE CASCADE,
  INDEX (amenity_id),
  INDEX (entity_type, entity_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- -----------------------------
-- MEDIA (Photos polymorphic)
-- -----------------------------
CREATE TABLE photo (
  id INT AUTO_INCREMENT PRIMARY KEY,
  entity_type VARCHAR(32) NOT NULL, -- hotel, room_type, room, review, etc.
  entity_id INT NOT NULL,
  url VARCHAR(1000) NOT NULL,
  caption VARCHAR(255),
  display_order INT DEFAULT 0,
  INDEX (entity_type, entity_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- -----------------------------
-- RATE PLANS
-- -----------------------------
CREATE TABLE rate_plan (
  id INT AUTO_INCREMENT PRIMARY KEY,
  hotel_id INT NOT NULL,
  code VARCHAR(64),
  name VARCHAR(255),
  prepayment_required TINYINT(1) DEFAULT 0,
  refundable TINYINT(1) DEFAULT 1,
  cancellation_policy_id VARCHAR(255),
  FOREIGN KEY (hotel_id) REFERENCES hotel(id) ON DELETE CASCADE,
  INDEX (hotel_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- -----------------------------
-- INVENTORY & LOCKS
-- -----------------------------
CREATE TABLE inventory (
  id INT AUTO_INCREMENT PRIMARY KEY,
  hotel_id INT NOT NULL,
  room_type_id INT NOT NULL,
  dt DATE NOT NULL,
  available_count INT DEFAULT 0,
  closed TINYINT(1) DEFAULT 0,
  FOREIGN KEY (hotel_id) REFERENCES hotel(id) ON DELETE CASCADE,
  FOREIGN KEY (room_type_id) REFERENCES room_type(id) ON DELETE CASCADE,
  UNIQUE KEY ux_inventory (hotel_id, room_type_id, dt),
  INDEX (hotel_id),
  INDEX (room_type_id),
  INDEX (dt)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE inventory_lock (
  id INT AUTO_INCREMENT PRIMARY KEY,
  hotel_id INT NOT NULL,
  room_type_id INT NOT NULL,
  dt DATE NOT NULL,
  locked_by VARCHAR(128),
  reason TEXT,
  FOREIGN KEY (hotel_id) REFERENCES hotel(id) ON DELETE CASCADE,
  FOREIGN KEY (room_type_id) REFERENCES room_type(id) ON DELETE CASCADE,
  UNIQUE KEY ux_invlock (hotel_id, room_type_id, dt),
  INDEX (dt)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- -----------------------------
-- BOOKINGS & PAYMENTS
-- -----------------------------
CREATE TABLE bookings (
  id INT AUTO_INCREMENT PRIMARY KEY,
  user_id INT DEFAULT NULL,
  hotel_id INT NOT NULL,
  booking_reference VARCHAR(64) NOT NULL UNIQUE,
  checkin_date DATE NOT NULL,
  checkout_date DATE NOT NULL,
  total_amount DECIMAL(12,2) NOT NULL,
  currency VARCHAR(8) DEFAULT 'USD',
  status VARCHAR(32) DEFAULT 'pending',
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (user_id) REFERENCES `user`(user_id) ON DELETE SET NULL,
  FOREIGN KEY (hotel_id) REFERENCES hotel(id) ON DELETE CASCADE,
  INDEX (user_id),
  INDEX (hotel_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE booking_room (
  id INT AUTO_INCREMENT PRIMARY KEY,
  booking_id INT NOT NULL,
  room_type_id INT DEFAULT NULL,
  rate_plan_id INT DEFAULT NULL,
  price_per_night DECIMAL(10,2) NOT NULL,
  FOREIGN KEY (booking_id) REFERENCES bookings(id) ON DELETE CASCADE,
  FOREIGN KEY (room_type_id) REFERENCES room_type(id) ON DELETE SET NULL,
  FOREIGN KEY (rate_plan_id) REFERENCES rate_plan(id) ON DELETE SET NULL,
  INDEX (booking_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE booking_payment (
  id INT AUTO_INCREMENT PRIMARY KEY,
  booking_id INT NOT NULL,
  provider_transaction_id VARCHAR(255),
  amount DECIMAL(12,2) NOT NULL,
  status VARCHAR(32) DEFAULT 'initiated',
  method VARCHAR(64),
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (booking_id) REFERENCES bookings(id) ON DELETE CASCADE,
  INDEX (booking_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE refund (
  id INT AUTO_INCREMENT PRIMARY KEY,
  booking_payment_id INT NOT NULL,
  amount DECIMAL(12,2) NOT NULL,
  reason TEXT,
  status VARCHAR(32) DEFAULT 'processing',
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (booking_payment_id) REFERENCES booking_payment(id) ON DELETE CASCADE,
  INDEX (booking_payment_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE invoice (
  id INT AUTO_INCREMENT PRIMARY KEY,
  booking_id INT NOT NULL,
  invoice_number VARCHAR(128),
  pdf_url VARCHAR(1000),
  issued_at DATETIME,
  total_amount DECIMAL(12,2),
  FOREIGN KEY (booking_id) REFERENCES bookings(id) ON DELETE CASCADE,
  INDEX (booking_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- -----------------------------
-- REVIEWS
-- -----------------------------
CREATE TABLE review (
  id INT AUTO_INCREMENT PRIMARY KEY,
  user_id INT DEFAULT NULL,
  hotel_id INT NOT NULL,
  rating TINYINT NOT NULL,
  title VARCHAR(255),
  body TEXT,
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  status VARCHAR(32) DEFAULT 'published',
  FOREIGN KEY (user_id) REFERENCES `user`(user_id) ON DELETE SET NULL,
  FOREIGN KEY (hotel_id) REFERENCES hotel(id) ON DELETE CASCADE,
  INDEX (hotel_id),
  INDEX (user_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- -----------------------------
-- SAVED LISTS (wishlist & compare)
-- -----------------------------
CREATE TABLE saved_list (
  id INT AUTO_INCREMENT PRIMARY KEY,
  user_id INT NOT NULL,
  name VARCHAR(255),
  list_type VARCHAR(32) DEFAULT 'wishlist', -- wishlist or compare
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (user_id) REFERENCES `user`(user_id) ON DELETE CASCADE,
  INDEX (user_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE saved_list_item (
  id INT AUTO_INCREMENT PRIMARY KEY,
  saved_list_id INT NOT NULL,
  hotel_id INT NOT NULL,
  added_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (saved_list_id) REFERENCES saved_list(id) ON DELETE CASCADE,
  FOREIGN KEY (hotel_id) REFERENCES hotel(id) ON DELETE CASCADE,
  INDEX (saved_list_id),
  INDEX (hotel_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- -----------------------------
-- SUPPORT
-- -----------------------------
CREATE TABLE support_ticket (
  id INT AUTO_INCREMENT PRIMARY KEY,
  user_id INT NOT NULL,
  subject VARCHAR(255),
  status VARCHAR(32) DEFAULT 'open',
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (user_id) REFERENCES `user`(user_id) ON DELETE CASCADE,
  INDEX (user_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE support_comment (
  id INT AUTO_INCREMENT PRIMARY KEY,
  ticket_id INT NOT NULL,
  user_id INT,
  comment TEXT,
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (ticket_id) REFERENCES support_ticket(id) ON DELETE CASCADE,
  FOREIGN KEY (user_id) REFERENCES `user`(user_id) ON DELETE SET NULL,
  INDEX (ticket_id),
  INDEX (user_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- -----------------------------
-- NOTIFICATIONS & PUSH TOKENS
-- -----------------------------
CREATE TABLE notification (
  id INT AUTO_INCREMENT PRIMARY KEY,
  user_id INT NOT NULL,
  type VARCHAR(64),
  title VARCHAR(255),
  body TEXT,
  data JSON,
  delivered TINYINT(1) DEFAULT 0,
  read_at DATETIME DEFAULT NULL,
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (user_id) REFERENCES `user`(user_id) ON DELETE CASCADE,
  INDEX (user_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE push_token (
  id INT AUTO_INCREMENT PRIMARY KEY,
  user_id INT NOT NULL,
  token VARCHAR(500) NOT NULL,
  platform VARCHAR(50),
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (user_id) REFERENCES `user`(user_id) ON DELETE CASCADE,
  INDEX (user_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- -----------------------------
-- OPERATIONS: housekeeping, shift_assignment, vendors, availability request
-- -----------------------------
CREATE TABLE housekeeping (
  id INT AUTO_INCREMENT PRIMARY KEY,
  room_id INT DEFAULT NULL,
  room_type_id INT DEFAULT NULL,
  date DATE,
  assigned_staff VARCHAR(128),
  status VARCHAR(64) DEFAULT 'pending',
  tasks JSON,
  FOREIGN KEY (room_id) REFERENCES room(id) ON DELETE SET NULL,
  FOREIGN KEY (room_type_id) REFERENCES room_type(id) ON DELETE SET NULL,
  INDEX (room_id),
  INDEX (room_type_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE shift_assignment (
  id INT AUTO_INCREMENT PRIMARY KEY,
  shift_name VARCHAR(128),
  staff_identifier VARCHAR(128),
  date DATE,
  notes TEXT
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE vendor (
  id INT AUTO_INCREMENT PRIMARY KEY,
  name VARCHAR(255),
  contact_info TEXT,
  contract_id VARCHAR(255),
  contract_start_date DATE,
  contract_end_date DATE,
  contract_terms TEXT
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE availability_req (
  id INT AUTO_INCREMENT PRIMARY KEY,
  user_id INT,
  hotel_id INT,
  date_from DATE,
  date_to DATE,
  num_rooms INT DEFAULT 1,
  status VARCHAR(32) DEFAULT 'open',
  FOREIGN KEY (user_id) REFERENCES `user`(user_id) ON DELETE SET NULL,
  FOREIGN KEY (hotel_id) REFERENCES hotel(id) ON DELETE CASCADE,
  INDEX (user_id),
  INDEX (hotel_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- -----------------------------
-- ANALYTICS, SETTINGS, SEO
-- -----------------------------
CREATE TABLE analytics_event (
  id INT AUTO_INCREMENT PRIMARY KEY,
  user_id INT DEFAULT NULL,
  event_type VARCHAR(128),
  metadata JSON,
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (user_id) REFERENCES `user`(user_id) ON DELETE SET NULL,
  INDEX (user_id),
  INDEX (event_type)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE setting (
  `key` VARCHAR(255) PRIMARY KEY,
  value JSON
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE seo_data (
  id INT AUTO_INCREMENT PRIMARY KEY,
  entity_type VARCHAR(64),
  entity_id INT,
  title VARCHAR(255),
  description TEXT,
  keywords VARCHAR(500),
  INDEX (entity_type, entity_id)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- -----------------------------
-- AUDIT
-- -----------------------------
CREATE TABLE audit_log (
  id INT AUTO_INCREMENT PRIMARY KEY,
  entity_type VARCHAR(128),
  entity_id INT,
  action VARCHAR(128),
  performed_by INT,
  payload JSON,
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  INDEX (performed_by),
  INDEX (entity_type, entity_id),
  FOREIGN KEY (performed_by) REFERENCES `user`(user_id) ON DELETE SET NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE booking_audit (
  id INT AUTO_INCREMENT PRIMARY KEY,
  booking_id INT NOT NULL,
  action VARCHAR(255),
  payload JSON,
  created_by INT,
  created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
  FOREIGN KEY (booking_id) REFERENCES bookings(id) ON DELETE CASCADE,
  FOREIGN KEY (created_by) REFERENCES `user`(user_id) ON DELETE SET NULL,
  INDEX (booking_id),
  INDEX (created_by)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
